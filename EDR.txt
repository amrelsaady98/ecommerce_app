Table Category {
  category_id int [pk, increment] // Primary Key
  name varchar
  description varchar
  icon_image image_field
  is_active boolean
  is_deleted boolean

Table Product {
  product_id int [pk, increment] // Primary Key
  code varchar
  cost_price decimal
  description varchar
  image1 varchar
  is_active boolean
  is_deleted boolean
  name varchar
  our_price decimal
  sale_price decimal
  stock_qty int
  year int
  category_id int [ref: > Category.category_id] // Foreign Key
}

Table Product_image {
  image varchar

  product_id int [ref: > Product.product_id]
